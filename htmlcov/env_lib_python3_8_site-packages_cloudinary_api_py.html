175def delete_derived_by_transformation(public_ids, transformations, 

176 resource_type='image', type='upload', invalidate=None, 

177 **options): 

178 """Delete derived resources of public ids, identified by transformations 

179 

180 :param public_ids: the base resources 

181 :type public_ids: list of str 

182 :param transformations: the transformation of derived resources, optionally including the format 

183 :type transformations: list of (dict or str) 

184 :param type: The upload type

187 :type resource_type: str

191 :rtype: dict 

192 """

197 "transformations": utils.build_eager(transformations), 

198 "keep_original": True}
